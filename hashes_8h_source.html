57static uint32_t hgen(const void *data, long length) {
58 register long i = length;
59 register uint32_t hv = 0;
60 register const unsigned char *s = (const unsigned char *)data;
61 while (i--) {
62 hv += *s++;
63 hv += (hv << 10);
64 hv ^= (hv >> 6);
65 }
66 hv += (hv << 3);
67 hv ^= (hv >> 11);
68 hv += (hv << 15);
69
70 return hv;
71}
